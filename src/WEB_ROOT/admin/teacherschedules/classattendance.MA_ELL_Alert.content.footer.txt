<script type = 'text/javascript'>
$j(document).ready(function() {

  var arrEllAlertCCIDS = new Array();
  arrEllAlertCCIDS.push(  
    {});

  // 'scrape' ccids from IDs within the attendance table
  const ccids = $j('#attendance-table .studentrow[id^="ccid_"]')
	                  // convert to IDs to integers (after removing the first 5 characters: 'ccid_') 
                    .map(function() { return parseInt(this.id.toString().slice(5,99),10); })
                     // join integers into a comma-seperated list (no quotes)
	                  .get().join(',');                                                       
  

  // Remove the last object from the array, if it's empty
  if ($j.isEmptyObject(arrEllAlertCCIDS[arrEllAlertCCIDS.length-1])) {
    arrEllAlertCCIDS.pop();
  }

  for (i=0; i<arrEllAlertCCIDS.length; i++) {
    var ps_alerts = $j('#ccid_' + arrEllAlertCCIDS[i].ccId + ' td.cvDemarcation:eq(1)');

    if (ps_alerts.length > 0) {
      var strPhloteAlertLink = "<a title='ELL/LEP/FLEP Alert' id='ma_ell_alert' href='/admin/alerts/MA_ELL_Alert.html?frn=001" +  arrEllAlertCCIDS[i].sDcid + "' class='dialogM'><img src='/images/img/MA_ELL_Alert.png' alt='ELL/LEP/FLEP Alert'/></a>";
      ps_alerts.append(strPhloteAlertLink);
    }
  }

});
</script>
